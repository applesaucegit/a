text = input("Input Text:") shift = int(input("Shift Value:")) result = "" for char in text: if char.isalpha(): if char.isupper(): result += chr(((ord(char)-65 + shift )%26)+65) else: result += chr(((ord(char)-97 + shift )%26)+97) else: result += char print("-"*30) print("Input Text:", text) print("Shift Value:", shift) print("Cipher Value:",result)

#3. SHA256 object

import hashlib password=input("Enter the password:") password_bytes=password.encode('utf-8') hashed_object= hashlib.sha256(password_bytes) output= hashed_object.hexdigest() print(output)
